Calculation Engine
// Pure Dart logic · no network calls · <1ms execution
Run Pace Calculator
Dart // Calculate pace from goal time + distance PaceResult calculateRunPace({ required double distanceKm, required Duration totalTime, }) { final totalMins = totalTime.inSeconds / 60; final pacePerKm = totalMins / distanceKm; final pacePerMile = pacePerKm * 1.60934; final speedKmh = distanceKm / (totalMins / 60); return PaceResult( pacePerKm: pacePerKm, // min/km pacePerMile: pacePerMile, // min/mi speedKmh: speedKmh, // km/h ); }
Running Distances
Dart enum RunDistance { fiveK('5K', 5.0), tenK('10K', 10.0), halfMarathon('Half Marathon', 21.0975), marathon('Marathon', 42.195); const RunDistance(this.label, this.km); final String label; final double km; }
Triathlon Total Time
Dart TriathlonResult calculateTriathlon({ required Duration swimTime, required Duration bikeTime, required Duration runTime, Duration t1 = Duration.zero, Duration t2 = Duration.zero, }) { final totalTime = swimTime + t1 + bikeTime + t2 + runTime; return TriathlonResult( totalTime: totalTime, swimSplit: swimTime, bikeSplit: bikeTime, runSplit: runTime, t1: t1, t2: t2, ); }
Race Distances
Dart enum TriDistance { sprint('Sprint', swim: 750, bikeKm: 20, runKm: 5), olympic('Olympic', swim: 1500, bikeKm: 40, runKm: 10), halfDistance('70.3', swim: 1900, bikeKm: 90, runKm: 21.1), fullDistance('Full Distance', swim: 3800, bikeKm: 180, runKm: 42.2); // NO "Ironman" — NEVER }

Storage Operations
Dart class HistoryService { final HistoryDao _dao; // Save a calculation Future<void> save(CalculationResult r) => _dao.insert(r.toEntity()); // List all, newest first Future<List> getAll() => _dao.findAllOrderedByDate(); // Delete one or all Future<void> delete(String id) => _dao.deleteById(id); Future<void> clearAll() => _dao.deleteAll(); }

Architecture Principles
Separation of concerns
Calculator logic is pure Dart with no Flutter dependencies — unit testable in isolation.
Service layer
HistoryService abstracts all database operations so screens never touch SQLite directly.
Reusable widgets
TimeInputWidget and ResultCard are shared across triathlon and run calculators.
Zero magic dependencies
No overengineered state management. Provider or simple setState for a calculator is sufficient and maintainable.
